```go
func NewTask(taskType string, payload map[string]interface{}) *Task {
    return &Task{
        ID:        uuid.New().String(),
        Type:      taskType,
        Payload:   payload,
        Status:    TaskPending,
        MaxRetry:  3,

// internal/queue/queue.go
type Queue struct {
    client *redis.Client
    name   string
}

func (q *Queue) Enqueue(ctx context.Context, t *task.Task) error {
    data, _ := t.ToJSON()
    return q.client.ZAdd(ctx, q.name, redis.Z{
        Score:  float64(-t.Priority),
        Member: string(data),
    }).Err()
}

func (q *Queue) Dequeue(ctx context.Context) (*task.Task, error) {
    result := q.client.ZPopMin(ctx, q.name)
    vals, _ := result.Result()
    if len(vals) == 0 {
        return nil, nil
    }
    return task.TaskFromJSON([]byte(vals[0].Member))
}

// cmd/server/main.go
func main() {
    client := redis.NewClient(&redis.Options{Addr: "localhost:6379"})
    q := queue.NewQueue(client, "tasks")

    r := gin.Default()
    r.POST("/tasks", func(c *gin.Context) {
        var req struct {
            Type     string                 `json:"type"`
            Payload  map[string]interface{} `json:"payload"`
            Priority int                    `json:"priority"`
        }
        c.ShouldBindJSON(&req)

        t := task.NewTask(req.Type, req.Payload)
        t.Priority = req.Priority
        q.Enqueue(c.Request.Context(), t)

        c.JSON(201, gin.H{"task_id": t.ID})
    })

    r.Run(":8080")
}
```

## 短链服务

```go
type ShortURL struct {
    ID        int        `json:"id"`
    ShortCode string     `json:"short_code"`
    LongURL   string     `json:"long_url"`
    Clicks    int        `json:"clicks"`
    ExpiresAt *time.Time `json:"expires_at"`
    CreatedAt time.Time  `json:"created_at"`
}

const charset = "abcdefghijklmnopqrstuvwxyzABCDEFGHIJKLMNOPQRSTUVWXYZ0123456789"

func generateShortCode(length int) string {
    b := make([]byte, length)
    for i := range b {
        b[i] = charset[rand.Intn(len(charset))]
    }
    return string(b)
}

type ShortURLHandler struct {
    store ShortURLStore
}

func (h *ShortURLHandler) Create(c *gin.Context) {
    var req struct {
        URL     string `json:"url" binding:"required,url"`
        Expires int    `json:"expires"`
    }
    c.ShouldBindJSON(&req)

    shortCode := generateShortCode(6)
    url := &ShortURL{
        ShortCode: shortCode,
        LongURL:   req.URL,
        Clicks:    0,
    }

    h.store.Create(url)
    c.JSON(201, gin.H{
        "short_code": shortCode,
        "short_url":  "http://localhost:8080/" + shortCode,
    })
}

func (h *ShortURLHandler) Redirect(c *gin.Context) {
    shortCode := c.Param("code")
    url, err := h.store.GetByShortCode(shortCode)
    if err != nil {
        c.JSON(404, gin.H{"error": "短链接不存在"})
        return
    }

    h.store.IncrementClicks(url.ID)
    c.Redirect(301, url.LongURL)
}
```

## WebSocket 实时推送

```go
type Client struct {
    conn *websocket.Conn
    send chan []byte
}

type Hub struct {
    clients    map[*Client]bool
    broadcast  chan []byte
    register   chan *Client
    unregister chan *Client
}

func (h *Hub) Run() {
    for {
        select {
        case client := <-h.register:
            h.clients[client] = true
        case client := <-h.unregister:
            delete(h.clients, client)
            close(client.send)
        case message := <-h.broadcast:
            for client := range h.clients {
                select {
                case client.send <- message:
                default:
                    close(client.send)
                    delete(h.clients, client)
                }
            }
        }
    }
}

var upgrader = websocket.Upgrader{
    CheckOrigin: func(r *http.Request) bool { return true },
}

func main() {
    hub := NewHub()
    go hub.Run()

    r := gin.Default()

    r.GET("/ws", func(c *gin.Context) {
        conn, _ := upgrader.Upgrade(c.Writer, c.Request, nil)
        client := &Client{conn: conn, send: make(chan []byte, 256)}
        hub.register <- client

        go func() {
            defer func() {
                hub.unregister <- client
                conn.Close()
            }()
            for {
                _, message, err := conn.ReadMessage()
                if err != nil {
                    break
                }
                hub.Broadcast(message)
            }
        }()
    })

    r.POST("/broadcast", func(c *gin.Context) {
        var req struct {
            Message string `json:"message"`
        }
        c.ShouldBindJSON(&req)
        hub.Broadcast([]byte(req.Message))
    })

    r.Run(":8080")
}
```

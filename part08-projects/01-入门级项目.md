## 命令行工具 (CLI)

```go
// cmd/todo/main.go
package main

import (
    "fmt"
    "os"
    "strconv"
)

type Todo struct {
    ID    int
    Title string
    Done  bool
}

type TodoList struct {
    items  []Todo
    nextID int
}

func NewTodoList() *TodoList {
    return &TodoList{items: make([]Todo, 0), nextID: 1}
}

func (t *TodoList) Add(title string) {
    t.items = append(t.items, Todo{ID: t.nextID, Title: title, Done: false})
    t.nextID++
    fmt.Println("✓ 添加成功")
}

func (t *TodoList) List() {
    if len(t.items) == 0 {
        fmt.Println("暂无任务")
        return
    }
    for _, item := range t.items {
        status := " "
        if item.Done {
            status = "✓"
        }
        fmt.Printf("[%s] %d. %s\n", status, item.ID, item.Title)
    }
}

func (t *TodoList) Done(id int) bool {
    for i, item := range t.items {
        if item.ID == id {
            t.items[i].Done = true
            return true
        }
    }
    return false
}

func main() {
    if len(os.Args) < 2 {
        fmt.Println("用法：todo <命令> [参数]")
        fmt.Println("命令：add, list, done, remove")
        os.Exit(1)
    }

    list := NewTodoList()
    command := os.Args[1]

    switch command {
    case "add":
        list.Add(os.Args[2])
    case "list":
        list.List()
    case "done":
        id, _ := strconv.Atoi(os.Args[2])
        list.Done(id)
    }
}
```

## Todo List API

```go
package main

import (
    "net/http"
    "sync"
    "github.com/gin-gonic/gin"
)

type Todo struct {
    ID    int    `json:"id"`
    Title string `json:"title"`
    Done  bool   `json:"done"`
}

type TodoStore struct {
    mu     sync.RWMutex
    todos  map[int]Todo
    nextID int
}

func NewTodoStore() *TodoStore {
    return &TodoStore{todos: make(map[int]Todo), nextID: 1}
}

var store = NewTodoStore()

func main() {
    r := gin.Default()

    r.GET("/todos", func(c *gin.Context) {
        store.mu.RLock()
        defer store.mu.RUnlock()
        c.JSON(200, gin.H{"data": store.todos})
    })

    r.POST("/todos", func(c *gin.Context) {
        var req struct {
            Title string `json:"title" binding:"required"`
        }
        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(400, gin.H{"error": err.Error()})
            return
        }

        store.mu.Lock()
        todo := Todo{ID: store.nextID, Title: req.Title, Done: false}
        store.todos[store.nextID] = todo
        store.nextID++
        store.mu.Unlock()

        c.JSON(201, gin.H{"data": todo})
    })

    r.Run(":8080")
}
```

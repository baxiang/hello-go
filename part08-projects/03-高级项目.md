## 实时数据推送系统

```go
// 使用 WebSocket + Redis Pub/Sub

type PubSub struct {
    redis *redis.Client
    hubs  map[string]*Hub  // 按频道分组
}

func (ps *PubSub) Subscribe(channel string, client *Client) {
    ps.hubs[channel].clients[client] = true

    // 订阅 Redis channel
    pubsub := ps.redis.Subscribe(context.Background(), channel)
    go func() {
        for msg := range pubsub.Channel() {
            client.send <- []byte(msg.Payload)
        }
    }()
}

func (ps *PubSub) Publish(channel string, data interface{}) {
    msg, _ := json.Marshal(data)
    ps.redis.Publish(context.Background(), channel, msg)
}
```

## 分布式锁服务

```go
// 基于 Redis 的分布式锁
type DistributedLock struct {
    client *redis.Client
    key    string
    value  string
}

func (l *DistributedLock) Acquire(ctx context.Context, ttl time.Duration) bool {
    return l.client.SetNX(ctx, l.key, l.value, ttl).Val()
}

func (l *DistributedLock) Release(ctx context.Context) bool {
    // 使用 Lua 脚本确保原子性
    script := redis.NewScript(`
        if redis.call("get", KEYS[1]) == ARGV[1] then
            return redis.call("del", KEYS[1])
        else
            return 0
        end
    `)
    return script.Run(ctx, l.client, []string{l.key}, l.value).Val() == 1
}

// 使用
lock := &DistributedLock{
    client: redis.NewClient(&redis.Options{Addr: "localhost:6379"}),
    key:    "lock:order:123",
    value:  uuid.New().String(),
}

if lock.Acquire(ctx, 10*time.Second) {
    defer lock.Release(ctx)
    // 执行临界区代码
}
```

## 配置中心

```go
// 基于 etcd 的配置中心
type ConfigCenter struct {
    client *clientv3.Client
    prefix string
}

func (c *ConfigCenter) Get(key string) (string, error) {
    resp, err := c.client.Get(context.Background(), c.prefix+key)
    if err != nil {
        return "", err
    }
    if len(resp.Kvs) == 0 {
        return "", errors.New("key not found")
    }
    return string(resp.Kvs[0].Value), nil
}

func (c *ConfigCenter) Watch(key string) <-chan string {
    ch := make(chan string)
    go func() {
        rch := c.client.Watch(context.Background(), c.prefix+key)
        for wresp := range rch {
            for _, ev := range wresp.Events {
                ch <- string(ev.Kv.Value)
            }
        }
    }()
    return ch
}
```

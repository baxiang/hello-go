# 3.5 并发安全与竞态检测

## 数据竞争 (Data Race)

### 什么是数据竞争

数据竞争发生在两个或多个 goroutine 同时访问同一内存位置，且至少有一个是写操作时。

```go
package main

import (
    "fmt"
    "sync"
)

var counter int  // 共享变量

func main() {
    var wg sync.WaitGroup

    // 启动多个 goroutine 修改 counter
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            counter++  // 数据竞争！
        }()
    }

    wg.Wait()
    fmt.Println("counter =", counter)  // 结果不确定
}
```

### 运行竞态检测

```bash
# 使用 -race 标志
go run -race main.go
go test -race ./...
go build -race main.go

# 输出示例
# ==================
# WARNING: DATA RACE
# Read at 0x00c000016098 by goroutine 7:
#   main.main.func1()
#       main.go:15
# Previous write at 0x00c000016098 by goroutine 6:
#   main.main.func1()
#       main.go:15
# ==================
```

## atomic 原子操作

### atomic 包基础

```go
package main

import (
    "fmt"
    "sync"
    "sync/atomic"
)

func main() {
    var counter atomic.Int64  // Go 1.19+

    var wg sync.WaitGroup
    for i := 0; i < 1000; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            counter.Add(1)  // 原子操作
        }()
    }

    wg.Wait()
    fmt.Println("counter =", counter.Load())  // 1000
}
```

### 原子类型 (Go 1.19+)

```go
package main

import (
    "fmt"
    "sync/atomic"
)

func main() {
    // 原子整数
    var int64Val atomic.Int64
    int64Val.Store(100)
    fmt.Println("Load:", int64Val.Load())

    result := int64Val.Add(50)  // 原子加法
    fmt.Println("Add:", result)

    swapped := int64Val.CompareAndSwap(150, 200)  // CAS
    fmt.Println("CAS:", swapped)

    old := int64Val.Swap(300)  // 交换
    fmt.Println("Swap old:", old)

    // 原子布尔值
    var boolVal atomic.Bool
    boolVal.Store(true)
    fmt.Println("Bool:", boolVal.Load())

    // 原子指针
    var ptrVal atomic.Pointer[string]
    s := "hello"
    ptrVal.Store(&s)
    fmt.Println("Pointer:", *ptrVal.Load())
}
```

### 传统原子函数

```go
package main

import (
    "fmt"
    "sync/atomic"
    "unsafe"
)

func main() {
    var val int64 = 100

    // Load
    v := atomic.LoadInt64(&val)
    fmt.Println("Load:", v)

    // Store
    atomic.StoreInt64(&val, 200)
    fmt.Println("Store 后:", val)

    // Add
    result := atomic.AddInt64(&val, 50)
    fmt.Println("Add:", result)

    // Swap
    old := atomic.SwapInt64(&val, 300)
    fmt.Println("Swap old:", old)

    // CompareAndSwap
    ok := atomic.CompareAndSwapInt64(&val, 300, 400)
    fmt.Println("CAS:", ok, "val:", val)

    // 原子指针操作
    var ptr unsafe.Pointer
    str := "hello"
    atomic.StorePointer(&ptr, unsafe.Pointer(&str))
    loaded := (*string)(atomic.LoadPointer(&ptr))
    fmt.Println("Pointer:", *loaded)
}
```

### atomic.Value 原子值

```go
package main

import (
    "fmt"
    "sync/atomic"
    "time"
)

type Config struct {
    MaxConn int
    Timeout time.Duration
}

func main() {
    var config atomic.Value
    config.Store(&Config{MaxConn: 100, Timeout: time.Second})

    // 读取配置 (无锁)
    go func() {
        for {
            c := config.Load().(*Config)
            fmt.Println("Config:", c.MaxConn)
            time.Sleep(100 * time.Millisecond)
        }
    }()

    // 更新配置
    time.Sleep(500 * time.Millisecond)
    config.Store(&Config{MaxConn: 200, Timeout: 2 * time.Second})

    time.Sleep(time.Second)
}
```

## sync.Map 并发安全 Map

### sync.Map 使用

```go
package main

import (
    "fmt"
    "sync"
)

func main() {
    var m sync.Map

    // 存储
    m.Store("name", "Alice")
    m.Store("age", 25)

    // 读取
    val, ok := m.Load("name")
    fmt.Println("name:", val, "exists:", ok)

    // 读取并删除
    val, ok = m.LoadAndDelete("age")
    fmt.Println("deleted:", val)

    // 读取或存储 (如果不存在)
    val, loaded := m.LoadOrStore("city", "Beijing")
    fmt.Println("city:", val, "loaded:", loaded)

    // 遍历
    m.Range(func(key, value interface{}) bool {
        fmt.Printf("%v: %v\n", key, value)
        return true  // 继续遍历
    })

    // 删除
    m.Delete("name")
}
```

### sync.Map vs map+Mutex

```go
package main

import (
    "sync"
    "testing"
)

// 方式 1: map + Mutex
type MutexMap struct {
    mu sync.RWMutex
    m  map[string]int
}

func NewMutexMap() *MutexMap {
    return &MutexMap{m: make(map[string]int)}
}

func (m *MutexMap) Get(key string) (int, bool) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    val, ok := m.m[key]
    return val, ok
}

func (m *MutexMap) Set(key string, val int) {
    m.mu.Lock()
    defer m.mu.Unlock()
    m.m[key] = val
}

// 方式 2: sync.Map
type SyncMap struct {
    m sync.Map
}

func (m *SyncMap) Get(key string) (int, bool) {
    val, ok := m.m.Load(key)
    if !ok {
        return 0, false
    }
    return val.(int), true
}

func (m *SyncMap) Set(key string, val int) {
    m.m.Store(key, val)
}

// 基准测试
func BenchmarkMutexMap(b *testing.B) {
    mm := NewMutexMap()
    b.RunParallel(func(pb *testing.PB) {
        i := 0
        for pb.Next() {
            mm.Set("key", i)
            mm.Get("key")
            i++
        }
    })
}

func BenchmarkSyncMap(b *testing.B) {
    var sm SyncMap
    b.RunParallel(func(pb *testing.PB) {
        i := 0
        for pb.Next() {
            sm.Set("key", i)
            sm.Get("key")
            i++
        }
    })
}

/*
Benchmark 结果 (读多写少场景):
BenchmarkSyncMap-8      约 30% 更快

Benchmark 结果 (写多读少场景):
BenchmarkMutexMap-8     约 20% 更快
*/
```

## 无锁编程基础

### 无锁栈示例

```go
package main

import (
    "sync/atomic"
    "unsafe"
)

type Node struct {
    val  int
    next *Node
}

type LockFreeStack struct {
    top unsafe.Pointer
}

func (s *LockFreeStack) Push(val int) {
    newNode := &Node{val: val}
    for {
        oldTop := atomic.LoadPointer(&s.top)
        newNode.next = (*Node)(oldTop)
        if atomic.CompareAndSwapPointer(&s.top, oldTop, unsafe.Pointer(newNode)) {
            break
        }
    }
}

func (s *LockFreeStack) Pop() (int, bool) {
    for {
        oldTop := atomic.LoadPointer(&s.top)
        if oldTop == nil {
            return 0, false
        }
        newTop := (*Node)(oldTop).next
        if atomic.CompareAndSwapPointer(&s.top, oldTop, unsafe.Pointer(newTop)) {
            return (*Node)(oldTop).val, true
        }
    }
}
```

### 无锁队列 (Ring Buffer)

```go
package main

import (
    "sync/atomic"
)

type RingBuffer struct {
    data     []int
    head     uint64  // 读位置
    tail     uint64  // 写位置
    capacity uint64
}

func NewRingBuffer(size int) *RingBuffer {
    return &RingBuffer{
        data:     make([]int, size),
        capacity: uint64(size),
    }
}

func (rb *RingBuffer) Push(val int) bool {
    tail := atomic.LoadUint64(&rb.tail)
    next := (tail + 1) % rb.capacity

    // 队列满
    if next == atomic.LoadUint64(&rb.head) {
        return false
    }

    rb.data[tail] = val
    atomic.StoreUint64(&rb.tail, next)
    return true
}

func (rb *RingBuffer) Pop() (int, bool) {
    head := atomic.LoadUint64(&rb.head)

    // 队列空
    if head == atomic.LoadUint64(&rb.tail) {
        return 0, false
    }

    val := rb.data[head]
    atomic.StoreUint64(&rb.head, (head+1)%rb.capacity)
    return val, true
}
```

## 并发安全模式

### 1. 使用 Mutex 保护共享状态

```go
package main

import (
    "sync"
)

type SafeCounter struct {
    mu    sync.Mutex
    count int
}

func (c *SafeCounter) Inc() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.count++
}

func (c *SafeCounter) Value() int {
    c.mu.Lock()
    defer c.mu.Unlock()
    return c.count
}
```

### 2. 使用 Channel 传递数据

```go
package main

import "fmt"

func main() {
    // 使用 channel 管理状态
    stateCh := make(chan int, 1)
    stateCh <- 0  // 初始状态

    // 多个 goroutine 通过 channel 修改状态
    for i := 0; i < 5; i++ {
        go func() {
            stateCh <- <-stateCh + 1
        }()
    }

    // 等待所有 goroutine 完成
    for i := 0; i < 5; i++ {
        fmt.Println("state:", <-stateCh)
        stateCh <- <-stateCh  // 放回 channel
    }
}
```

### 3. 使用 sync.Once 单次初始化

```go
package main

import (
    "fmt"
    "sync"
)

var (
    instance *Singleton
    once     sync.Once
)

type Singleton struct {
    value int
}

func GetInstance() *Singleton {
    once.Do(func() {
        instance = &Singleton{value: 42}
        fmt.Println("初始化单例")
    })
    return instance
}

func main() {
    var wg sync.WaitGroup
    for i := 0; i < 5; i++ {
        wg.Add(1)
        go func() {
            defer wg.Done()
            s := GetInstance()
            fmt.Printf("获取实例：%p\n", s)
        }()
    }
    wg.Wait()
}
```

### 4. 使用 RWMutex 优化读多写少场景

```go

import (
```

## 竞态检测实战

### 检测和修复数据竞争

```go
package main

import (
    "fmt"
    "sync"
)

type SafeMap struct {
    mu   sync.RWMutex
    data map[string]int
}

func NewSafeMap() *SafeMap {
    return &SafeMap{
        data: make(map[string]int),
    }
}

func (m *SafeMap) Get(key string) (int, bool) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    val, ok := m.data[key]
    return val, ok
}

func (m *SafeMap) Set(key string, val int) {
    m.mu.Lock()
    defer m.mu.Unlock()
    m.data[key] = val
}

func (m *SafeMap) Delete(key string) {
    m.mu.Lock()
    defer m.mu.Unlock()
    delete(m.data, key)
}

func (m *SafeMap) Range(f func(string, int) bool) {
    m.mu.RLock()
    defer m.mu.RUnlock()
    for k, v := range m.data {
        if !f(k, v) {
            break
        }
    }
}

func main() {
    m := NewSafeMap()

    var wg sync.WaitGroup
    for i := 0; i < 100; i++ {
        wg.Add(3)

        // 写操作
        go func(i int) {
            defer wg.Done()
            m.Set(fmt.Sprintf("key%d", i), i)
        }(i)

        // 读操作
        go func(i int) {
            defer wg.Done()
            m.Get(fmt.Sprintf("key%d", i))
        }(i)

        // 遍历操作
        go func() {
            defer wg.Done()
            m.Range(func(k string, v int) bool {
                return true
            })
        }()
    }

    wg.Wait()
}
```

## 并发安全检查清单

```
并发安全检查清单:

[ ] 使用 go run -race 检测数据竞争
[ ] 共享变量使用 atomic 或 Mutex 保护
[ ] 避免在循环中共享循环变量
[ ] channel 使用后及时关闭 (发送方)
[ ] select 中处理 channel 关闭情况
[ ] goroutine 有正确的退出机制
[ ] 使用 context 控制 goroutine 生命周期
[ ] 避免 goroutine 泄露
[ ] 优先使用值传递代替指针传递
[ ] 文档化并发安全的 API
[ ] 使用 sync.Once 进行单次初始化
[ ] 读多写少场景使用 RWMutex
[ ] 考虑使用 channel 代替共享内存
[ ] 测试覆盖并发场景
[ ] 基准测试验证性能

常见陷阱:

1. 循环变量共享
   for i := 0; i < 10; i++ {
       go func() {
           fmt.Println(i)  // 错误：所有 goroutine 看到相同的 i
       }()
   }

   // 正确做法:
   for i := 0; i < 10; i++ {
       i := i  // 创建局部副本
       go func() {
           fmt.Println(i)
       }()
   }

2. 未等待 goroutine 完成
   for i := 0; i < 10; i++ {
       go func() { doSomething() }()
   }
   // 程序可能在这些 goroutine 完成前退出

   // 正确做法:
   var wg sync.WaitGroup
   for i := 0; i < 10; i++ {
       wg.Add(1)
       go func() {
           defer wg.Done()
           doSomething()
       }()
   }
   wg.Wait()

3. Mutex 未释放
   mu.Lock()
   doSomething()  // 如果 panic，锁永远不会释放
   mu.Unlock()

   // 正确做法:
   mu.Lock()
   defer mu.Unlock()
   doSomething()
```

## 性能对比

```go
package main

import (
    "sync"
    "sync/atomic"
    "testing"
)

// 方案 1: Mutex
type MutexCounter struct {
    mu    sync.Mutex
    count int64
}

func (c *MutexCounter) Inc() {
    c.mu.Lock()
    defer c.mu.Unlock()
    c.count++
}

// 方案 2: atomic
type AtomicCounter struct {
    count atomic.Int64
}

func (c *AtomicCounter) Inc() {
    c.count.Add(1)
}

// 基准测试
func BenchmarkMutexCounter(b *testing.B) {
    c := &MutexCounter{}
    b.RunParallel(func(pb *testing.PB) {
        for pb.Next() {
            c.Inc()
        }
    })
}

func BenchmarkAtomicCounter(b *testing.B) {
    c := &AtomicCounter{}
    b.RunParallel(func(pb *testing.PB) {
        for pb.Next() {
            c.Inc()
        }
    })
}

/*
典型结果:
BenchmarkAtomicCounter-8    比 Mutex 快约 50-100%
原因：
- atomic 使用 CPU 原子指令 (CAS)
- Mutex 涉及系统调用和上下文切换
- 低竞争时 atomic 优势明显
- 高竞争时两者性能接近
*/
```

# 3.5 并发模式

## Worker Pool

```go
package main

import (
    "fmt"
    "sync"
    "time"
)

type Job struct {
    ID   int
    Data string
}

type Result struct {
    JobID  int
    Output string
}

func worker(id int, jobs <-chan Job, results chan<- Result, wg *sync.WaitGroup) {
    defer wg.Done()
    for job := range jobs {
        fmt.Printf("Worker %d 处理任务 %d\n", id, job.ID)
        time.Sleep(100 * time.Millisecond)
        results <- Result{JobID: job.ID, Output: "完成"}
    }
}

func main() {
    const numWorkers = 3
    const numJobs = 10

    jobs := make(chan Job, numJobs)
    results := make(chan Result, numJobs)

    var wg sync.WaitGroup

    // 启动 worker
    for i := 1; i <= numWorkers; i++ {
        wg.Add(1)
        go worker(i, jobs, results, &wg)
    }

    // 发送任务
    for j := 1; j <= numJobs; j++ {
        jobs <- Job{ID: j, Data: fmt.Sprintf("task-%d", j)}
    }
    close(jobs)

    // 等待完成
    go func() {
        wg.Wait()
        close(results)
    }()

    // 收集结果
    for r := range results {
        fmt.Printf("结果：%d - %s\n", r.JobID, r.Output)
    }
}
```

## Producer-Consumer

```go
package main

import (
    "fmt"
    "sync"
)

func main() {
    ch := make(chan int, 100)
    var wg sync.WaitGroup

    // 生产者
    wg.Add(1)
    go func() {
        defer wg.Done()
        for i := 0; i < 10; i++ {
            ch <- i
        }
        close(ch)
    }()

    // 消费者
    wg.Add(1)
    go func() {
        defer wg.Done()
        for v := range ch {
            fmt.Println("消费:", v)
        }
    }()

    wg.Wait()
}
```

## Fan-In/Fan-Out

```go
package main

import (
    "fmt"
    "sync"
)

// Fan-Out
func fanOut(input <-chan int, numWorkers int) []<-chan int {
    outputs := make([]<-chan int, numWorkers)
    for i := 0; i < numWorkers; i++ {
        out := make(chan int)
        outputs[i] = out
        go func(ch chan<- int, idx int) {
            defer close(ch)
            for v := range input {
                if v%numWorkers == idx {
                    ch <- v
                }
            }
        }(out, i)
    }
    return outputs
}

// Fan-In
func fanIn(channels ...<-chan int) <-chan int {
    out := make(chan int)
    var wg sync.WaitGroup
    wg.Add(len(channels))

    for _, ch := range channels {
        go func(c <-chan int) {
            defer wg.Done()
            for v := range c {
                out <- v
            }
        }(ch)
    }

    go func() {
        wg.Wait()
        close(out)
    }()

    return out
}
```

## Pipeline

```go
package main

import "fmt"

func stage1(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            out <- n * 2
        }
    }()
    return out
}

func stage2(in <-chan int) <-chan int {
    out := make(chan int)
    go func() {
        defer close(out)
        for n := range in {
            out <- n + 10
        }
    }()
    return out
}

func main() {
    input := make(chan int, 10)
    for i := 1; i <= 10; i++ {
        input <- i
    }
    close(input)

    // 连接管道
    p1 := stage1(input)
    p2 := stage2(p1)

    // 输出结果
    for result := range p2 {
        fmt.Println(result)
    }
}
```

## ErrGroup

```go
package main

import (
    "context"
    "fmt"
    "golang.org/x/sync/errgroup"
    "time"
)

func main() {
    // 基础使用
    var g errgroup.Group
    for i := 0; i < 3; i++ {
        i := i
        g.Go(func() error {
            fmt.Printf("任务 %d\n", i)
            if i == 2 {
                return fmt.Errorf("任务 %d 失败", i)
            }
            return nil
        })
    }
    if err := g.Wait(); err != nil {
        fmt.Println("错误:", err)
    }

    // 带 Context
    ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
    defer cancel()

    g, ctx := errgroup.WithContext(ctx)
    for i := 0; i < 5; i++ {
        i := i
        g.Go(func() error {
            select {
            case <-ctx.Done():
                return ctx.Err()
            default:
                time.Sleep(time.Second)
                return nil
            }
        })
    }

    // 限制并发数
    g.SetLimit(3)
}
```
